{

    void init_I() { for (int i = 1; i <= n; i++) b[i][i] = 1; }

        for (int i = 1; i <= n; i++)

    }

{
    T res; res.init_I();
    while (y)

        x = x * x; y >>= 1;
    }
    return res;
